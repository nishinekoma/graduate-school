C_values = [0.001, 0.01, 0.1, 1.0, 10.0, 100.0]
results_c = []

for C in C_values:
  lr_c = LogisticRegression(C=C, max_iter=1000, random_state=42)
  lr_c.fit(X_train_sc, y_train)
  pred = lr_c.predict(X_test_sc)
  rep = classification_report(pred, y_test, output_dict=True, zero_division=0)
  results_c.append({
      "C": C,"テスト正解率": round(accuracy_score(y_test, pred), 3),
      "SH再現率": round(rep.get("1", {}).get("recall", 0), 3),
      "SH F1": round(f1_score(y_test, pred, zero_division=0), 3),
      "非ゼロ係数数": int(np.sum(lr_c.coef_[0] != 0)),
  })
c_df = pd.DataFrame(results_c)

def get_metrics(y_true, y_pred, name):
  rep = classification_report(y_true, y_pred, output_dict=True, zero_division=0)
  return {
      "手法": name,
      "正解率": round(accuracy_score(y_true, y_pred), 3),
      "SH適合率": round(rep["1"]["precision"], 3),
      "SH再現率": round(rep["1"]["recall"], 3),
      "SH F1": round(rep["1"]["f1-score"], 3),
      "macro F1": round(rep["macro avg"]["f1-score"], 3),
      }

all_results = [
    get_metrics(y_test, y_pred_dt, "決定木（DT）"),
    get_metrics(y_test, y_pred_rf, "ランダムフォレスト（RF）"),
    get_metrics(y_test, y_pred_xgb, "XGBoost"),
    get_metrics(y_test, y_pred_lr, "ロジスティック回帰（LR）"),
    get_metrics(y_test, y_pred_svm, "SVM（線形）"),
]
